\begin{frame}
	
	\begin{block}{Definición}
	La \alert{matríz Laplaciana} es la matriz (con las filas y columnas indexadas con los vértices de $G$), tal que la entrada $uv$ esta definida por
	\[
		L(G)_{u,v}=
		\begin{cases}
		deg_G (v) & \text{ si } u=v,\\
		-m_{u,v} & \text{ en otro caso.}
		\end{cases}
	\]
	donde $m_{u,v}$ es el número de aristas entre $u$ y $v$. 
	\end{block}

	\begin{block}{Regla del desbordamiento}
		Sea $G=(V,E)$ una multigráfica y sea ${\bf a}$ una configuración.
		Si algún vértice $v$ es tal que $deg_G(v)\leq a_v$, entonces \alert{desbordaremos} el vértice $v$ al restar $deg_G(v)$ a $a_v$ y a cada vecino $u$ de $v$ sumamos $m_{v,u}$.
	\end{block}

\begin{frame}
	\begin{block}{Definición}
		El \alert{grupo de pilas de arena} de $G$ es el conjunto de configuraciones recurrentes. Y lo denotaremos por $SP(G,s)$.
	\end{block}

	\begin{block}{Teorema (Burning Algorithm)}
		Sea $G$ una multigráfica con $s\in V(G)$ un vertice distinguido.
		Además, sea $\beta=L(G)_s$.
		Una configuración $u$ es recurrente si y solo si $s(u+\beta)=u$.
	\end{block}
\end{frame}


\begin{frame}
	
	{\bf ¿Cuántas configuraciones recurrentes hay?}
	
	\begin{block}{Definición}
		Sea $G$ una multigráfica.
		Una subgráfica \alert{generadora} es una subgráfica $G'$ de $G$ con $V(G')=V(G)$.
	\end{block}
	
	\begin{block}{Teorema matriz árbol}
		El número de árboles generadores de una gráfica $G$ es igual a cualquier menor $n-1\times n-1$ de $L(G)$. 
	\end{block}

	El \redalert{número de configuraciones recurrentes} es igual al \redalert{número de árboles generadores.}

\end{frame}


\begin{frame}


	Dada una configuración $c$, definimos 
	\[
		nivel(c)=\sum_{v} c_v.
	\]
	
	\begin{block}{Teorema}
		Sea $G=(V,E)$ una gráfica con sink $s$ y $c$ una configuración recurrente, entonces
		\[
			| E |-\deg_G (s)  \leq \redalert{nivel(c)} \leq 2| E |-\deg_G (s)-|V|+1
		\]
	\end{block}
	
\end{frame}

\begin{frame}	

	Para $i\geq 0$, tomemos \redalert{$n_i$} como el número de configuraciones críticas con nivel \redalert{$i+|E|-\deg(s)$.}
	
	La \redalert{función generadora} de las configuraciones críticas es el polinomio
	\[
		P(G;y)=\sum_{i=0}^{|E|-|V|+1} n_i y^i
	\]

\end{frame}


\begin{frame}	

	\begin{block}{Definición}
		Sea $G$ una multigráfica.
		El polinomio de Tutte de $G$ es el polinomio $T(G;x,y) \in \mathbb{Z}[x,y]$ definido como sigue:
		\[
			T(G;x,y)=
			\begin{cases}
				1 & \text{ si } E(G)=\emptyset,\\
				xT(G\slash e) & \text{ si } e \text{ es un puente},\\
				yT(G\backslash e;x,y) & \text{ si } e \text{ es un lazo},\\
				T(G\slash e;x,y) + T(G\backslash e; x,y) & \text{ si } e \text{ no es ni lazo ni puente}.
			\end{cases}
		\]
	\end{block}
	
	Por ejemplo $T(K_4;x,y)=3x^2+4xy+3y^2+2x+2y+y^3+x^3$

\end{frame}


\begin{frame}

	\begin{block}{Teorema [Merino,1997]}
		La {\bf función generadora} de las configuraciones recurrentes es el {\bf polinomio de Tutte} a lo largo de la linea $x=1$,
		\[
			P(G,y)=T(G;1,y).
		\]
	\end{block}
	
	$P(K_4,y)=6+6y+3y^2+y^3$.

\end{frame}


\begin{frame}
	
	\Titulo{Descripción Algebraica (Grupo Crítico)}

	La descripción algebraica del grupo de pilas de arena también es conocida como el grupo crítico.\\
	
	\vspace{.5cm}

	El \redalert{grupo crítico} de $G$, denotado por $K(G)$, esta dado por
	\[
		K(G)=\mathbb{Z}^{n-1}/L(G,s)^{t},
	\]
	donde $L(G,s)$ es la matriz Laplaciana reducida.
\end{frame}


\begin{frame}
	
	\begin{block}{Teorema}
		El grupo de pilas de arena de $G$ junto con la operación $\oplus=s( \cdot + \cdot )$ es un grupo abeliano fínito.
	\end{block}

	\begin{block}{Proposición}
		Para toda multigráfica $G$ el grupo de pilas de arena $SP(G,s)\cong K(G)$ para todo $s\in V(G)$.
	\end{block}
	{\tiny (relación matriz laplaciana)}

\end{frame}

	\begin{enumerate}
		\item El grupo crítico se puede obtener calculando la \redalert{forma normal de Smith} de la matriz Laplaciana reducida.
	\end{enumerate}

\begin{frame}

	\begin{block}{Teorema [Cori y Rossin 2001]}
	Si $K_n$ es la gráfica completa de $n$ vértices. 
	Entonces
	\[
		K(K_n)\cong \bigoplus_{i=1}^{n-2} \mathbb{Z}_n.
	\]
	\end{block}
	\begin{proof}
		\[
			L(K_n,s)=
			\left[
			\begin{array}{cccc}
				n-1 & -1  & \cdots & -1 \\
				-1  & n-1 & \cdots & -1 \\
				\vdots & \vdots & \ddots & \vdots \\
				-1  &  -1 & \cdots & n-1 \\
			\end{array}
			\right]
			\sim
			\left[
			\begin{array}{cccc}
				1 & 0  & \cdots & 0 \\
				0  & n & \cdots & 0 \\
				\vdots & \vdots & \ddots & \vdots \\
				0  & 0 & \cdots & n \\
			\end{array}
			\right]
		\]
	\end{proof}

\end{frame}




\begin{frame}

	\Titulo{¿Cómo calcular el grupo de pilas de arena?}

	\begin{enumerate}
		\item El grupo crítico se puede obtener calculando la \redalert{forma normal de Smith} de la matriz Laplaciana reducida.
		\item El grupo de pilas de arena se necesita calcular la configuración de desbordado (burning algorithm) y calcular combinatoriamente las configuraciones recurrentes.
	\end{enumerate}
	
\end{frame}

\begin{frame}
	\Titulo{Programación Líneal Entera y el grupo de pilas de arena.}

	\begin{block}{Teorema}
		Sea $G$ una multigráfica, $s\in V(G)$, ${\bf 0} \leq {\bf c} \leq {\bf d}_{(G,s)}-{\bf 1}$ una configuración estable de $G$ y ${\bf x}^*$ una solución óptima del siguiente problema líneal entero:
		\begin{eqnarray}
			\text{\rm maximizar }   & & |{\bf x}| \nonumber\\
			\text{\rm sujeto a: } & & {\bf 0}\leq L(G,s)^t{\bf x}+{\bf c} \leq {\bf d}_{(G,s)}-{\bf 1}\\
            			           & & {\bf x}\geq {\bf 0}\nonumber,
		\end{eqnarray}
		entonces $L(G,s)^t{\bf x}^* + {\bf c}\in SP(G,s)$ y $[{\bf c}]=[L(G,s)^t{\bf x}^* + {\bf c}]$.
	\end{block}
\end{frame}


\begin{frame}

	\begin{block}{Ejemplo}
	Sea $c=(0,0,1,0)$ una configuración en $(C_5,v_5)$. 
	El correspondiente programa líneal entero es:
	\begin{eqnarray*}
		\text{\rm maximizar } & & x_1+x_2+x_3+x_4 \\
		\text{\rm sujeto a: } & & \\
		& &
			\mathbf{0}
		\leq
		\left[
			\begin{array}{cccc}
				2 & -1 & 0 & 0 \\
				-1 & 2 & -1 & 0 \\
				0 & -1 & 2 & -1 \\
				0 & 0 & -1 & 2 \\
			\end{array}
		\right]
		\mathbf{x}
		+
		\left[
			\begin{array}{c}
				0 \\
				0 \\
				1 \\
				0 \\
			\end{array}
		\right]
		\leq
		\left[
			\begin{array}{c}
				1 \\
				1 \\
				1 \\
				1 \\
			\end{array}
		\right]\\
	& & \mathbf{x}\geq \mathbf{0}
	\end{eqnarray*}
	Obtenemos que $\mathbf{x}^*=(1,1,1,1)$. 
	Y $L(G,s)^t{\bf x}^* + {\bf c}=(1,0,1,1)$ es una configuración recurrente.
\end{block}

\end{frame}


\begin{frame}

	\begin{block}{Corolario}
		Sea $G$ una multigráfica, ${\bf x}^*$ una solución óptima del siguiente programa líneal entero:
\begin{eqnarray}
\text{\rm maximizar }   & & |{\bf x}| \nonumber\\
\text{\rm sujeto a: } & & {\bf 0}\leq L(G,s)^t{\bf x} \leq {\bf d}_{(G,s)}-{\bf 1}\\
                       & & {\bf x}\geq {\bf 0}\nonumber,
\end{eqnarray}
entonces $L(G,s)^t{\bf x}^* \in SP(G,s)$ es la identidad de $SP(G,s)$.
\end{block}

\begin{block}{Corolario}
	Sea $G$ una multigráfica $r$-regular, entonces $r{\bf 1}$ es la identidad del grupo de pilas de arena, $SP(c(G),s)$. 
\end{block}

\end{frame}


\begin{frame}
	\Titulo{La Familia $\mathcal G_i$}
	
	Un resultado clásico (ver [Jacobson,1985]) es que
	\[
		K(G)\cong \mathbb{Z}_{d_1} \oplus \mathbb{Z}_{d_2} \oplus \cdots \mathbb{Z}_{d_r},
	\]
	donde $d_i> 0$ y $d_i \mid d_j$ si $i\leq j$.\\
	Los enteros $d_1, \dots, d_r$ son llamados \redalert{factores invariantes}.\\
	Más aún, $d_i=\Delta_i/ \Delta_{i-1}$, donde $\Delta_0=1$ y $\Delta_i$ es el gcd de los $i\times i$ menores de $L(G)$. 

\end{frame}


\begin{frame}
	
	Denotemos por:
	\begin{itemize}
		\item $\redalert{f_i(G)}$, el número de factores invariantes iguales a $i$ de $K(G)$ y
		\item $\redalert{f_{\geq 2}(G)}$ al número de factores invariantes mayores a 2 de $K(G)$.
	\end{itemize}
	
	Se sabe (ver [Godsil, 2001]) que 
	\[
		f_1(G)+f_{\geq 2}(G)=n-c,
	\]
	donde $c$ es el número de componentes conexas.
	
	\begin{block}{Definición}
		Denotemos por $\redalert{\mathcal G_i}$ a la familia de gráficas \redalert{ simples conexas} con $\redalert{f_1(G)=i}$.
	\end{block}
	
\end{frame}


\begin{frame}
	
	\begin{block}{Ejemplo}
		Por ejemplo, la siguiente gráfica pertenece a $\redalert{\mathcal G_2}$.

\begin{frame}	
	
	\begin{block}{Observación}
		El conjunto $\mathcal G_i$ no es cerrado bajo subgráficas inducidas.\\
		Por ejemplo, el cono de $S_3$ pertenece a $\mathcal G_2$, pero $S_3$ pertenece a $\mathcal G_3$.\\
		También, la gráfica $K_6\setminus M_2$ pertenece a $\mathcal G_3$, y $K_5\setminus M_2$ pertenece a $\mathcal G_2$.
	\end{block}
	
	\begin{block}{Teorema [Lorenzini, 1989]}
		Si $G$ es una gráfica \redalert{simple conexa}, entonces los siguientes enunciados son equivalentes:
		\begin{enumerate}[i.]
			\item $G \in \mathcal{G}_1$,
			\item $G$ es $P_2$-free,
			\item $G$ es la gráfica completa.
		\end{enumerate}
	\end{block}
	
\end{frame}
