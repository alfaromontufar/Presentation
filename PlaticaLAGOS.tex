\section{Grupo Crítico}


\subsection*{}


\begin{frame}
	
	\frametitle{Grupo Crítico}
	
	En la literatura el grupo de pilas de arena tambien es conocido como el grupo crítico, el grupo Jacobiano, el grupo de Picard, etc.\\
	\vspace{.5cm}
	Más aún, existen varias formas equivalentes de definirlo.\\
	\vspace{.5cm}
	La descripción algebraica del grupo de pilas de arena también es conocida como el grupo crítico.
	
\end{frame}


\begin{frame}

	\begin{block}{Definición}
		El \redalert{grupo crítico $K(G)$}  de $G$ está definido como la parte de torción del cokernel de la matriz Laplaciana de $G$.
		\[
			coker(L(G))= \mathbb{Z}^{n}/{\sf Im}L(G) = \mathbb{Z}\oplus \alert{K(G)}.
		\]
	\end{block}

	\begin{block}{Proposición}
		Para toda gráfica $G$, y todo $s\in V(G)$, tenemos \[SP(G,s)\cong K(G).\]
	\end{block}
	{\tiny (relación matriz laplaciana)}

\end{frame}


\subsection{Factores invariantes}


\begin{frame}

	\frametitle{¿Cómo calcular el grupo de pilas de arena?}

	El grupo crítico se puede obtener calculando la \redalert{forma normal de Smith} de la matriz Laplaciana.
	
	\pause
	
	\begin{block}{Definición}
		Dos matrices $M$ y $N$ son \alert{equivalentes} si existen $P,Q\in GL_n(\mathbb{Z})$ tal que $N=PMQ$.  
	\end{block}
	
	\pause
	
	\begin{block}{Teorema}
		Si $M$ y $N$ son equivalentes, entonces $\mathbb{Z}^{n}/M \cong \mathbb{Z}^{n}/N$.
	\end{block}
	
	\pause
	
	\begin{block}{Definición}
		La \alert{forma normal de Smith} de una matriz $M$ es la matriz diagonal \alert{${\sf diag}(d_1, ..., d_r)$}, equivalente a $A$, que satisface $d_i> 0$ y $d_i \mid d_j$ si $i\leq j$.
	\end{block}
	
\end{frame}


\begin{frame}
	
	\begin{block}{Teorema}
		\[
			K(G)\cong \mathbb{Z}_{d_1} \oplus \mathbb{Z}_{d_2} \oplus \cdots \oplus \mathbb{Z}_{d_r},
		\]
		donde $d_i> 0$ y $d_i \mid d_j$ si $i\leq j$.
	\end{block}
	
	\begin{block}{Definición}
		Los enteros $d_1, \dots, d_r$ son llamados \redalert{factores invariantes}.
	\end{block}
	
	\begin{block}{Proposición}
		$d_i=\Delta_i/ \Delta_{i-1}$, donde $\Delta_0=1$ y $\Delta_i$ es el gcd de los $i\times i$ menores de $L(G)$.
	\end{block}

\end{frame}


\begin{frame}

	\begin{block}{Teorema [Cori y Rossin 2001]}
	Si $K_n$ es la gráfica completa de $n$ vértices. 
	Entonces
	\[
		K(K_n)\cong \bigoplus_{i=1}^{n-2} \mathbb{Z}_n.
	\]
	\end{block}
	\begin{proof}
		\[
			L(K_n,s)=
			\left[
			\begin{array}{cccc}
				n-1 & -1  & \cdots & -1 \\
				-1  & n-1 & \cdots & -1 \\
				\vdots & \vdots & \ddots & \vdots \\
				-1  &  -1 & \cdots & n-1 \\
			\end{array}
			\right]
			\sim
			\left[
			\begin{array}{cccc}
				1 & 0  & \cdots & 0 \\
				0  & n & \cdots & 0 \\
				\vdots & \vdots & \ddots & \vdots \\
				0  & 0 & \cdots & n \\
			\end{array}
			\right]
		\]
	\end{proof}

\end{frame}

\begin{frame}

	\begin{block}{Teorema}
		Sea $\mathcal{C}_{n}$ el ciclo gordo con $m_i$ la multiplicidad entre $v_i$ y $v_{i+1}$, entonces 
		\[
			K(\mathcal{C}_{n})=\mathbb{Z}_{\Delta_1} \oplus \left(\bigoplus_{i=2}^{n-1} \mathbb{Z}_{\Delta_i/ \Delta_{i-1}}\right).
	\]
		con 
		\[
			\Delta_i=
			\begin{cases}
				\gcd\left\{m_{j_1} \cdots m_{j_i} \right\}_{1\leq j_1 < \cdots < j_i \leq n} & \text{si } 1 \leq i \leq n-2, \\
				(-1)^{n} \sum_{i=1}^{n} m_1 m_2 \cdots \hat{m_i} \cdots m_n & \text{si } i=n-1. \\ 
			\end{cases}
		\]
	\end{block}

\end{frame}


\begin{frame}

	\begin{block}{Corolario}
		Sea $\mathcal{C}_{n}$ el ciclo gordo con $m_i=m$, entonces 
		\[
			K(\mathcal{C}_{n})=\left(\bigoplus_{i=1}^{n-2} \mathbb{Z}_m\right)\oplus\mathbb{Z}_{mn}.
		\]
	\end{block}


	\begin{block}{Corolario}
		El grupo de pilas de arena del ciclo de $n$ vertices es isomorfo a $\mathbb{Z}_n$.
	\end{block}

\end{frame}


\subsection{Gráficas con conectividad uno.}


\begin{frame}

	\frametitle{Gráficas con conectividad uno.}
	
	\begin{block}{Teorema}
		Sea $G$ una gráfica y sean $G_1$, $G_2$, ..., $G_l$ sus bloques, entonces
		\[
			K(G)=K(G_1)\oplus K(G_2) \oplus \cdots \oplus K(G_l).
		\]
	\end{block}

\end{frame}


\begin{frame}

	\begin{block}{Pregunta}
		Dado un grupo fínito abeliano $\Gamma$. ¿Existe una gráfica $G$ tal que $K(G)=\Gamma$?
	\end{block}

\begin{frame}
	\Titulo{Algunos isomorfismos.}
	
	\begin{block}{Teorema [Cori y Rossin, 2000]}
		Sea $G$ una gráfica y $G^*$ su dual, entonces $K(G)\cong K(G^*)$.
	\end{block}

	\begin{block}{Teorema [Wagner, 2000]}
		Sean $G$ y $H$ gráficas. 
		Si las matroides gráficas de $G$ y $H$ son isomorfas, entonces $K(G)\cong K(H)$.
	\end{block}
	
\end{frame}

\subsection{La Familia $\mathcal G_i$}


\begin{frame}
	\frametitle{La Familia $\mathcal G_i$}
	
	\begin{block}{}
		Denotemos por:
		\begin{itemize}
			\item $\redalert{f_i(G)}$, el número de factores invariantes iguales a $i$ de $K(G)$,
			\item $\redalert{f_{\geq 2}(G)}$ al número de factores invariantes mayores a 2 de $K(G)$.
		\end{itemize}
	\end{block}
	
	\begin{block}{Teorema [Godsil, 2001]}
	\[
		f_1(G)+f_{\geq 2}(G)=n-c,
	\]
	donde $c$ es el número de componentes conexas.
	\end{block}
	
	\begin{block}{Corolario}
		Una gráfica simple conexa cumple
		\[
		1\leq f_1(G) \text{     y     }  f_{\geq 2}(G)\leq n-2.
		\]
	\end{block}
	
\end{frame}

	\begin{block}{Definición}
		Denotemos por $\redalert{\mathcal G_i}$ a la familia de gráficas \redalert{ simples conexas} con $\redalert{f_1(G)=i}$.

\begin{frame}
	\begin{block}{Pregunta}
		¿Qué tan frecuente es cíclico el grupo crítico? es decir, ¿Qué tan frecuente $f_1(G)$ es igual a $n-1$ o $n-2$?
	\end{block}
	
	\begin{block}{Conjetura [D. Wagner, 2001]}
		Casi todas las gráficas simples y conexas tienen grupo crítico ciclico.
	\end{block}
	
	\begin{block}{Teorema [M. Wood, 2014]}
		La probabilidad de que el grupo crítico de una gráfica aleatoria sea cíclica es asintóticamente a lo más 
		\[
			\zeta(3)^{-1}\zeta(5)^{-1}\zeta(7)^{-1}\zeta(9)^{-1}\zeta(11)^{-1} \cdots \approx 0.7935212
		\]
		donde $\zeta$ es la función zeta de Riemann.
	\end{block}
\end{frame}

	\begin{block}{Teorema [Lorenzini, 1989]}
		Si $G$ es una gráfica \redalert{simple conexa}, entonces los siguientes enunciados son equivalentes:
		\begin{enumerate}[i.]
			\item $G \in \mathcal{G}_1$,
			\item $G$ es $P_2$-libre,
			\item $G$ es la gráfica completa.
		\end{enumerate}
	\end{block}
	
\end{frame}


\begin{frame}

	\begin{block}{Pregunta}
		¿Qué podemos decir sobre $\mathcal G_2$ y $\mathcal G_3$?
	\end{block}
	
	\begin{block}{Teorema}
		Sea $G$ una gráfica \redalert{simple conexa}.
		Entonces, $G\in \mathcal{G}_2$ si y solamente si $G$ es una de las siguientes gráficas:
		\begin{enumerate}[i.]
			\item $K_{n_1,n_2,n_3}$, donde $n_1$, $n_2$ y $n_3$ tienen la misma paridad.
			\item $L_{n_1,n_2,n_3}$, si $n_1, n_2, n_3 \geq 3$ tienen la misma paridad, u otros once casos.
